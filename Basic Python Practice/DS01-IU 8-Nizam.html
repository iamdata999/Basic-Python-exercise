import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

#In this code, we load the 'tips' dataset from Seaborn using the
#sns.load_dataset() function. We create the box plot using
#sns.boxplot() and specify the 'day' column as the x-axis and the 
#'total_bill' column as the y-axis.

#We add labels to the x-axis and y-axis using plt.xlabel() and
#plt.ylabel(), respectively. The title of the plot is set using plt.title().
#plt.show() to display the box plot.



# Load the 'tips' dataset from Seaborn
data = sns.load_dataset('tips')

# Create the box plot
sns.boxplot(x='day', y='total_bill', data=data)

# Add labels and title
plt.xlabel('Day')
plt.ylabel('Total Bill')
plt.title('Box Plot of Total Bill across Days')

# Display the plot
plt.show()

# Calculate the correlation matrix

# Create the Heatmap

# Add title
